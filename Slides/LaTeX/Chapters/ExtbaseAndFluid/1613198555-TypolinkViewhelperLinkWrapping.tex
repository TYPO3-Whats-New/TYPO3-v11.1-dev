% Feature | 92338 | Allow link text wrapping in TypolinkViewhelper

\begin{frame}[fragile]
	\frametitle{Extbase et Fluid}
	\framesubtitle{TypolinkViewhelper}

	\lstset{basicstyle=\fontsize{7}{9}\ttfamily}

	\begin{itemize}
		\item Le ViewHelper \texttt{<f:link.typolink ...>} fourni l'argument
			\texttt{textWrap} qui entoure le titre du lien.

		\item Par exemple~:
\begin{lstlisting}
<f:link.typolink parameter="123" textWrap="<span>|</span>" />
\end{lstlisting}

		\item Sortie~:
\begin{lstlisting}
<a href="page-123"><span>My page title</span></a>
\end{lstlisting}

		\item Note~: n'oubliez pas de prot√©ger les guillemets~:
\begin{lstlisting}
<f:link.typolink parameter="123" textWrap="<span class=\"small\">|</span>" />
\end{lstlisting}

	\end{itemize}

\end{frame}
